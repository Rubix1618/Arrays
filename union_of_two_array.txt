//union of two array using hashset

class Solution {
    public int[] unionArray(int[] nums1, int[] nums2) {
        int start1=0;
        int start2=0;
        HashSet<Integer> res=new HashSet<>();
        for(int ele:nums1){
            res.add(ele);
        }
        for(int ele:nums2){
            res.add(ele);
        }
        int [] ans=new int[res.size()];
        int count=0;
        for(int i:res){
            ans[count++]=i;
        }
        return ans;
        
    }
}
