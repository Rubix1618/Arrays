class Solution {
    public int longestSubarray(int[] nums, int k) {
       int l=0;
       int r=0;
       int sum=0;
       int max=0;
       while(r<nums.length){
        sum+=nums[r];
        if(sum==k){
            max=Math.max(max,r-l+1);
        }
        if(sum>k){
            sum-=nums[l];
            l++;
        }
        r++;

       }
       return max;
    }
}
