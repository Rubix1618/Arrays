// finding the missing number
class Solution {
    public int missingNumber(int[] nums) {
        int len =nums.length;
        int actual_sum=0;
        for(int i=0;i<=len;i++){
            actual_sum+=i;
        }
        for(int ele:nums){
            actual_sum-=ele;
        }
        return actual_sum;
    }
}
